<script>
// ============ STATE ============
let currentUser = null;
let currentRoom = null;
let mediaRecorder = null;
let audioChunks = [];
let isRecording = false;
let localStream = null;
let recordedBlobs = {};

// ============ LOGIN ============
function login() {
const name = document.getElementById('loginName').value.trim();
if (!name) {
alert('Enter your name!');
return;
}
currentUser = name;
document.getElementById('loginScreen').style.display = 'none';
document.getElementById('mainApp').style.display = 'flex';
document.getElementById('userAvatar').textContent = name.charAt(0).toUpperCase();
showRoomModal();
}

function logout() {
location.reload();
}

// ============ ROOMS ============
function showRoomModal() {
document.getElementById('roomModal').classList.add('active');
loadRooms();
}

function closeRoomModal() {
document.getElementById('roomModal').classList.remove('active');
}

function loadRooms() {
const rooms = JSON.parse(localStorage.getItem('chatRooms') || '["General", "Gaming", "Music", "Random"]');
const container = document.getElementById('roomList');
container.innerHTML = '';

rooms.forEach(room => {
const div = document.createElement('div');
div.className = 'room-item';
div.textContent = '🏠 ' + room;
div.onclick = () => joinRoom(room);
container.appendChild(div);
});
}

function createRoom() {
const name = document.getElementById('newRoomName').value.trim();
if (!name) return;

let rooms = JSON.parse(localStorage.getItem('chatRooms') || '["General", "Gaming", "Music", "Random"]');
if (!rooms.includes(name)) {
rooms.push(name);
localStorage.setItem('chatRooms', JSON.stringify(rooms));
}

joinRoom(name);
}

function joinRoom(roomName) {
currentRoom = roomName;
closeRoomModal();

// Show chat area
document.getElementById('emptyState').style.display = 'none';
document.getElementById('chatHeader').style.display = 'flex';
document.getElementById('chatMessages').style.display = 'block';
document.getElementById('inputArea').style.display = 'flex';

document.getElementById('headerName').textContent = roomName;
updateUserCount();

// Load messages for THIS ROOM ONLY
loadRoomMessages();

// Update sidebar with active room
updateSidebar();
}

function updateSidebar() {
const container = document.getElementById('chatList');
container.innerHTML = '';

const rooms = JSON.parse(localStorage.getItem('chatRooms') || '["General", "Gaming", "Music", "Random"]');

rooms.forEach(room => {
const msgs = JSON.parse(localStorage.getItem('room_' + room) || '[]');
const lastMsg = msgs.length > 0 ? msgs[msgs.length - 1] : null;

const div = document.createElement('div');
div.className = 'chat-item' + (room === currentRoom ? ' active' : '');
div.onclick = () => joinRoom(room);

div.innerHTML = &lt;div class="chat-avatar"&gt;🏠&lt;/div&gt; &lt;div class="chat-info"&gt; &lt;div class="chat-header-row"&gt; &lt;span class="chat-name"&gt;${room}</span>
<span class="chat-time">${lastMsg ? lastMsg.time : ''}&lt;/span&gt; &lt;/div&gt; &lt;div class="chat-header-row"&gt; &lt;span class="chat-message"&gt;${lastMsg ? lastMsg.text.substring(0, 30) : 'No messages'}</span>
</div>
</div>
`;
container.appendChild(div);
});
}

function updateUserCount() {
// Simulate user count
const count = Math.floor(Math.random() * 10) + 2;
document.getElementById('userCount').textContent = count;
}

// ============ MESSAGES (PER ROOM) ============
function loadRoomMessages() {
const container = document.getElementById('chatMessages');
container.innerHTML = '';

const msgs = JSON.parse(localStorage.getItem('room_' + currentRoom) || '[]');
msgs.forEach(msg => displayMessage(msg));
}

function displayMessage(msg) {
const container = document.getElementById('chatMessages');
const isMe = msg.name === currentUser;

const div = document.createElement('div');
div.className = 'message ' + (isMe ? 'sent' : 'received');

let content = '';

if (msg.type === 'voice') {
const blobUrl = recordedBlobs[msg.blobId] || msg.audioUrl;
content = &lt;div class="voice-msg"&gt; &lt;button class="voice-play" onclick="playVoice('${blobUrl}')">▶</button>
<div class="voice-bar"><div class="voice-fill" style="width:60%"></div></div>
<span class="voice-time">0:03</span>
</div>
`;
} else {
content = '<strong>' + msg.name + '</strong><br>' + msg.text;
}

div.innerHTML = content + '<div class="message-time">' + msg.time + '</div>';
container.appendChild(div);
container.scrollTop = container.scrollHeight;
}

function sendMessage() {
const input = document.getElementById('messageInput');
const text = input.value.trim();
if (!text || !currentRoom) return;

const time = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

const msg = {
name: currentUser,
text: text,
time: time,
type: 'text'
};

// Save to THIS ROOM ONLY
let msgs = JSON.parse(localStorage.getItem('room_' + currentRoom) || '[]');
msgs.push(msg);
localStorage.setItem('room_' + currentRoom, JSON.stringify(msgs));

displayMessage(msg);
input.value = '';
updateSidebar();

// Fake reply from someone else
if (Math.random() > 0.5) {
setTimeout(() => {
const replies = ['Nice!', '👍', 'Hello!', 'How are you?', 'Cool!', 'Interesting...'];
const reply = {
name: 'User' + Math.floor(Math.random() * 100),
text: replies[Math.floor(Math.random() * replies.length)],
time: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
type: 'text'
};

let roomMsgs = JSON.parse(localStorage.getItem('room_' + currentRoom) || '[]');
roomMsgs.push(reply);
localStorage.setItem('room_' + currentRoom, JSON.stringify(roomMsgs));

if (document.getElementById('chatMessages').style.display !== 'none') {
displayMessage(reply);
updateSidebar();
}
}, 2000);
}
}

// ============ VOICE MESSAGES ============
async function startVoiceRecord() {
if (isRecording) {
stopRecording();
return;
}

try {
const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
mediaRecorder = new MediaRecorder(stream);
audioChunks = [];

mediaRecorder.ondataavailable = (e) => {
audioChunks.push(e.data);
};

mediaRecorder.onstop = () => {
const audioBlob = new Blob(audioChunks, { type: 'audio/webm' });
const audioUrl = URL.createObjectURL(audioBlob);
const blobId = 'blob_' + Date.now();
recordedBlobs[blobId] = audioUrl;

// Send voice message
const time = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
const msg = {
name: currentUser,
text: 'Voice message',
time: time,
type: 'voice',
blobId: blobId,
audioUrl: audioUrl
};

let msgs = JSON.parse(localStorage.getItem('room_' + currentRoom) || '[]');
msgs.push(msg);
localStorage.setItem('room_' + currentRoom, JSON.stringify(msgs));

displayMessage(msg);
updateSidebar();

stream.getTracks().forEach(track => track.stop());
};

mediaRecorder.start();
isRecording = true;
document.getElementById('recordingIndicator').style.display = 'inline';
document.getElementById('recordBtn').textContent = '⏹';

} catch (err) {
alert('Microphone access denied or not available');
}
}

function stopRecording() {
if (mediaRecorder && isRecording) {
mediaRecorder.stop();
isRecording = false;
document.getElementById('recordingIndicator').style.display = 'none';
document.getElementById('recordBtn').textContent = '🎤';
}
}

function playVoice(url) {
const audio = new Audio(url);
audio.play();
}

// ============ VIDEO CALLS ============
async function startVideoCall() {
try {
localStream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });
document.getElementById('localVideo').srcObject = localStream;
document.getElementById('videoOverlay').classList.add('active');

// In a real app, you'd connect to another user via WebRTC
// For demo, we just show local camera
alert('Video call started! (Demo mode - showing your camera only. In production, this would connect to another user.)');

} catch (err) {
alert('Camera access denied or not available');
}
}

async function startVoiceCall() {
try {
localStream = await navigator.mediaDevices.getUserMedia({ audio: true });
document.getElementById('videoOverlay').classList.add('active');
document.getElementById('remoteVideo').style.display = 'none';

alert('Voice call started! (Demo mode)');

} catch (err) {
alert('Microphone access denied');
}
}

function endCall() {
if (localStream) {
localStream.getTracks().forEach(track => track.stop());
localStream = null;
}
document.getElementById('videoOverlay').classList.remove('active');
document.getElementById('remoteVideo').style.display = 'block';
}

function toggleMute() {
if (localStream) {
localStream.getAudioTracks().forEach(track => {
track.enabled = !track.enabled;
});
}
}

function toggleVideo() {
if (localStream) {
localStream.getVideoTracks().forEach(track => {
track.enabled = !track.enabled;
});
}
}

// ============ UTILITIES ============
function toggleEmoji() {
const emojis = ['😀', '😂', '🥰', '😎', '🤔', '👍', '❤️', '🔥', '🎉', '👏'];
const input = document.getElementById('messageInput');
input.value += emojis[Math.floor(Math.random() * emojis.length)];
}

document.getElementById('messageInput').addEventListener('keypress', (e) => {
if (e.key === 'Enter') sendMessage();
});

document.getElementById('searchInput').addEventListener('input', function() {
const search = this.value.toLowerCase();
document.querySelectorAll('.chat-item').forEach(item => {
const name = item.querySelector('.chat-name').textContent.toLowerCase();
item.style.display = name.includes(search) ? 'flex' : 'none';
});
});
